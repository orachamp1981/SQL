CREATE TABLE EMP_LOG AS SELECT * FROM EMP WHERE ROWNUM = 0;
ALTER TABLE EMP_LOG ADD (REMARKS VARCHAR2(2000), TABLE_NAME VARCHAR2(200), ACTIVITY_DATE DATE DEFAULT SYSDATE);

CREATE OR REPLACE TRIGGER EMP_LOG
AFTER INSERT OR UPDATE OR DELETE ON EMP
REFERENCING NEW AS NEW OLD AS OLD FOR EACH ROW
DECLARE
A VARCHAR2(2000);
BEGIN
 IF INSERTING 
 THEN
 INSERT INTO EMP_LOG
 VALUES
 (:NEW.EMPNO, :NEW.ENAME, :NEW.JOB, :NEW.MGR, :NEW.HIREDATE, :NEW.SAL, :NEW.COMM, :NEW.DEPTNO, 'AFTER INSERT', 'EMP', SYSDATE);
 END IF;
 IF UPDATING
 THEN
  INSERT INTO EMP_LOG
 VALUES
 (:OLD.EMPNO, :OLD.ENAME, :OLD.JOB, :OLD.MGR, :OLD.HIREDATE, :OLD.SAL, :OLD.COMM, :OLD.DEPTNO, 'BEFORE INSERT', 'EMP', SYSDATE);
 INSERT INTO EMP_LOG
 VALUES
 (:NEW.EMPNO, :NEW.ENAME, :NEW.JOB, :NEW.MGR, :NEW.HIREDATE, :NEW.SAL, :NEW.COMM, :NEW.DEPTNO, 'AFTER INSERT', 'EMP', SYSDATE);
 END IF;

 END;
